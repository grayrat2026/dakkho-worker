   164‚Üí   164‚Üí/** Convert a page name to a URL path, including dynamic params */
   165‚Üí   165‚Üíexport function pageToUrl(page: string, params?: Record<string, unknown>): string {
   166‚Üí   166‚Üí  let path = pageToPath[page] || `/${page}`;
   167‚Üí   167‚Üí
   168‚Üí   168‚Üí  // Append dynamic params as URL segments
   169‚Üí   169‚Üí  if (params) {
   170‚Üí   170‚Üí    const extraSegments: string[] = [];
   171‚Üí   171‚Üí    if (params.courseId) extraSegments.push(String(params.courseId));
   172‚Üí   172‚Üí    if (params.videoId) extraSegments.push(String(params.videoId));
   173‚Üí   173‚Üí    if (params.instructorId) extraSegments.push(String(params.instructorId));
   174‚Üí   174‚Üí    if (params.query) extraSegments.push(encodeURIComponent(String(params.query)));
   175‚Üí   175‚Üí    if (params.userId) extraSegments.push(String(params.userId));
   176‚Üí   176‚Üí    if (extraSegments.length > 0) {
   177‚Üí   177‚Üí      path += '/' + extraSegments.join('/');
   178‚Üí   178‚Üí    }
   179‚Üí   179‚Üí  }
   180‚Üí   180‚Üí
   181‚Üí   181‚Üí  return path;
   182‚Üí   182‚Üí}

   184‚Üí   184‚Üí/** Convert a URL path back to a page name and params */
   185‚Üí   185‚Üíexport function urlToPage(urlPath: string): { page: string; params: Record<string, string> } {
   186‚Üí   186‚Üí  // Normalize: remove trailing slash
   187‚Üí   187‚Üí  const normalized = urlPath.replace(/\/+$/, '') || '/';
   188‚Üí   188‚Üí
   189‚Üí   189‚Üí  // Try exact match first
   190‚Üí   190‚Üí  const exactMatch = pathToPage[normalized];
   191‚Üí   191‚Üí  if (exactMatch) {
   192‚Üí   192‚Üí    return { page: exactMatch, params: {} };
   193‚Üí   193‚Üí  }
   194‚Üí   194‚Üí
   195‚Üí   195‚Üí  // Try matching with dynamic segments
   196‚Üí   196‚Üí  // e.g., /course/detail/c1 ‚Üí { page: 'course-detail', params: { courseId: 'c1' } }
   197‚Üí   197‚Üí  // e.g., /search/web%20dev ‚Üí { page: 'search', params: { query: 'web dev' } }
   198‚Üí   198‚Üí  // e.g., /instructor/profile/ins1 ‚Üí { page: 'instructor-profile', params: { instructorId: 'ins1' } }
   199‚Üí   199‚Üí
   200‚Üí   200‚Üí  const segments = normalized.split('/').filter(Boolean);
   201‚Üí   201‚Üí
   202‚Üí   202‚Üí  // Try progressively shorter paths
   203‚Üí   203‚Üí  for (let i = segments.length; i >= 1; i--) {
   204‚Üí   204‚Üí    const prefix = '/' + segments.slice(0, i).join('/');
   205‚Üí   205‚Üí    const match = pathToPage[prefix];
   206‚Üí   206‚Üí    if (match) {
   207‚Üí   207‚Üí      const extraSegments = segments.slice(i);
   208‚Üí   208‚Üí      const params: Record<string, string> = {};
   209‚Üí   209‚Üí
   210‚Üí   210‚Üí      // Assign extra segments to known param keys based on page type
   211‚Üí   211‚Üí      if (match === 'course-detail' && extraSegments[0]) params.courseId = extraSegments[0];
   212‚Üí   212‚Üí      else if (match === 'video-player' && extraSegments[0]) params.videoId = extraSegments[0];
   213‚Üí   213‚Üí      else if (match === 'instructor-profile' && extraSegments[0]) params.instructorId = extraSegments[0];
   214‚Üí   214‚Üí      else if (match === 'search' && extraSegments[0]) params.query = decodeURIComponent(extraSegments[0]);
   215‚Üí   215‚Üí      else if (match === 'profile' && extraSegments[0]) params.userId = extraSegments[0];
   216‚Üí   216‚Üí      else {
   217‚Üí   217‚Üí        // Generic: store as param0, param1, etc.
   218‚Üí   218‚Üí        extraSegments.forEach((seg, idx) => {
   219‚Üí   219‚Üí          params[`param${idx}`] = seg;
   220‚Üí   220‚Üí        });
   221‚Üí   221‚Üí      }
   222‚Üí   222‚Üí
   223‚Üí   223‚Üí      return { page: match, params };
   224‚Üí   224‚Üí    }
   225‚Üí   225‚Üí  }
   226‚Üí   226‚Üí
   227‚Üí   227‚Üí  // No match found ‚Üí 404
   228‚Üí   228‚Üí  return { page: 'error-404', params: {} };
   229‚Üí   229‚Üí}

   232‚Üí   232‚Üíinterface NavigationState {
   233‚Üí   233‚Üí  currentPage: Page;
   234‚Üí   234‚Üí  pageParams: Record<string, unknown>;
   235‚Üí   235‚Üí  navigate: (page: Page, params?: Record<string, unknown>) => void;
   236‚Üí   236‚Üí  goBack: () => void;
   237‚Üí   237‚Üí  history: Array<{ page: Page; params: Record<string, unknown> }>;
   238‚Üí   238‚Üí  sidebarOpen: boolean;
   239‚Üí   239‚Üí  toggleSidebar: () => void;
   240‚Üí   240‚Üí  setSidebarOpen: (open: boolean) => void;
   241‚Üí   241‚Üí  syncFromUrl: (urlPath: string) => void;
   242‚Üí   242‚Üí}

   246‚Üí   246‚Üíexport const useNavigationStore = create<NavigationState>((set, get) => ({
   247‚Üí   247‚Üí  currentPage: 'home',
   248‚Üí   248‚Üí  pageParams: {},
   249‚Üí   249‚Üí  history: [],
   250‚Üí   250‚Üí  sidebarOpen: false,
   251‚Üí   251‚Üí  toggleSidebar: () => set((s) => ({ sidebarOpen: !s.sidebarOpen })),
   252‚Üí   252‚Üí  setSidebarOpen: (open) => set({ sidebarOpen: open }),
   253‚Üí   253‚Üí  navigate: (page, params = {}) => {
   254‚Üí   254‚Üí    const state = get();
   255‚Üí   255‚Üí    const url = pageToUrl(page, params);
   256‚Üí   256‚Üí
   257‚Üí   257‚Üí    // Update browser URL
   258‚Üí   258‚Üí    if (typeof window !== 'undefined') {
   259‚Üí   259‚Üí      isNavigatingProgrammatically = true;
   260‚Üí   260‚Üí      window.history.pushState({ page, params }, '', url);
   261‚Üí   261‚Üí      // Reset the flag after a microtask
   262‚Üí   262‚Üí      queueMicrotask(() => { isNavigatingProgrammatically = false; });
   263‚Üí   263‚Üí    }
   264‚Üí   264‚Üí
   265‚Üí   265‚Üí    set({
   266‚Üí   266‚Üí      currentPage: page,
   267‚Üí   267‚Üí      pageParams: params,
   268‚Üí   268‚Üí      history: [...state.history, { page: state.currentPage, params: state.pageParams }],
   269‚Üí   269‚Üí    });
   270‚Üí   270‚Üí  },
   271‚Üí   271‚Üí  goBack: () => {
   272‚Üí   272‚Üí    const state = get();
   273‚Üí   273‚Üí    if (typeof window !== 'undefined') {
   274‚Üí   274‚Üí      window.history.back();
   275‚Üí   275‚Üí    }
   276‚Üí   276‚Üí    if (state.history.length > 0) {
   277‚Üí   277‚Üí      const prev = state.history[state.history.length - 1];
   278‚Üí   278‚Üí      set({
   279‚Üí   279‚Üí        currentPage: prev.page,
   280‚Üí   280‚Üí        pageParams: prev.params,
   281‚Üí   281‚Üí        history: state.history.slice(0, -1),
   282‚Üí   282‚Üí      });
   283‚Üí   283‚Üí    }
   284‚Üí   284‚Üí  },
   285‚Üí   285‚Üí  syncFromUrl: (urlPath: string) => {
   286‚Üí   286‚Üí    const { page, params } = urlToPage(urlPath);
   287‚Üí   287‚Üí    if (page !== get().currentPage || JSON.stringify(params) !== JSON.stringify(get().pageParams)) {
   288‚Üí   288‚Üí      set({
   289‚Üí   289‚Üí        currentPage: page as Page,
   290‚Üí   290‚Üí        pageParams: params,
   291‚Üí   291‚Üí      });
   292‚Üí   292‚Üí    }
