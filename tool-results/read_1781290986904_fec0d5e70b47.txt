     1→'use client';
     2→
     3→import { useState, useMemo } from 'react';
     4→import { motion, AnimatePresence } from 'framer-motion';
     5→import {
     6→  Plus, Trash2, BookOpen, FolderOpen,
     7→  Video, Radio, FileText, StickyNote, ChevronDown,
     8→  Link as LinkIcon, Edit3, Eye, X, Save,
     9→  MessageSquare, Send, Users,
    10→} from 'lucide-react';
    11→import { toast } from 'sonner';
    12→import { GlassCard } from '@/components/shared/GlassCard';
    13→import { GradientButton } from '@/components/shared/GradientButton';
    14→import { LoadingSkeleton } from '@/components/shared/LoadingSkeleton';
    15→import { EmptyState } from '@/components/shared/EmptyState';
    16→import {
    17→  useCourseSubjects, useCurriculum, useRemoveSubjectFromCourse, useCourse,
    18→  useCreateChapter, useDeleteChapter,
    19→  useCreateLesson, useDeleteLesson,
    20→  useReviews, useReplyReview,
    21→} from '@/lib/api-hooks';
    22→import { useNavigationStore } from '@/lib/store';
    23→import { cn, formatDurationShort, relativeTime } from '@/lib/utils';
    24→import { staggerChildren, collapseExpand } from '@/lib/animations';
    25→
    26→type AddMode = 'chapter' | 'lesson' | null;
    27→
    28→const lessonTypeIcon: Record<string, typeof Video> = {
    29→  video: Video, link: LinkIcon, youtube: Video, document: FileText,
    30→};
    31→const lessonTypeLabel: Record<string, string> = {
    32→  video: 'Video', link: 'Link', youtube: 'YouTube', document: 'Document/PDF',
    33→};
    34→const lessonTypeColor: Record<string, string> = {
    35→  video: 'bg-neutral-100 text-neutral-700 dark:bg-neutral-800/30 dark:text-neutral-300',
    36→  link: 'bg-blue-100 text-blue-700 dark:bg-blue-900/30 dark:text-blue-400',
    37→  youtube: 'bg-red-100 text-red-700 dark:bg-red-900/30 dark:text-red-400',
    38→  document: 'bg-amber-100 text-amber-700 dark:bg-amber-900/30 dark:text-amber-400',
    39→};
    40→
    41→export function CourseCurriculum() {
    42→  const params = useNavigationStore((s) => s.params);
    43→  const navigate = useNavigationStore((s) => s.navigate);
    44→  const courseId = params.courseId || '';
    45→
    46→  const { data: courseData, loading: courseLoading } = useCourse(courseId);
    47→  const { data: subjectsData, loading: subjectsLoading, refetch: refetchSubjects } = useCourseSubjects(courseId);
    48→  const { data: curriculumData, loading: curriculumLoading, refetch: refetchCurriculum } = useCurriculum(courseId);
    49→  const { mutate: removeSubject } = useRemoveSubjectFromCourse(courseId);
    50→  const { mutate: createChapter, loading: creatingChapter } = useCreateChapter(courseId);
    51→  const { mutate: deleteChapter, loading: deletingChapter } = useDeleteChapter(courseId);
    52→  const { mutate: createLesson, loading: creatingLesson } = useCreateLesson(courseId);
    53→  const { mutate: deleteLesson, loading: deletingLesson } = useDeleteLesson(courseId);
    54→
    55→  // Reviews
    56→  const { data: reviewsData, refetch: refetchReviews } = useReviews(1);
    57→  const { mutate: replyReview, loading: replyingReview } = useReplyReview();
    58→
    59→  const course = courseData?.course || courseData;
    60→  const subjects = useMemo(() => subjectsData?.subjects || [], [subjectsData]);
    61→  const curriculum = curriculumData?.curriculum || { subjects: [], chapters: [], lessons: [], resources: [] };
    62→  const reviews = useMemo(() => reviewsData?.reviews || [], [reviewsData]);
    63→
    64→  // Expand/collapse state
    65→  const [expandedSubjects, setExpandedSubjects] = useState<Set<string>>(new Set());
    66→  const [expandedChapters, setExpandedChapters] = useState<Set<string>>(new Set());
    67→
    68→  // Inline form state
    69→  const [addMode, setAddMode] = useState<AddMode>(null);
    70→  const [addSubjectId, setAddSubjectId] = useState<string>('');
    71→  const [addChapterId, setAddChapterId] = useState<string>('');
    72→
    73→  // Chapter form
    74→  const [chapterTitle, setChapterTitle] = useState('');
    75→  const [chapterDesc, setChapterDesc] = useState('');
    76→
    77→  // Lesson form
    78→  const [lessonTitle, setLessonTitle] = useState('');
    79→  const [lessonType, setLessonType] = useState<'video' | 'link' | 'youtube' | 'document'>('video');
    80→  const [lessonVideoUrl, setLessonVideoUrl] = useState('');
    81→  const [lessonDesc, setLessonDesc] = useState('');
    82→  const [lessonIsPreview, setLessonIsPreview] = useState(false);
    83→
    84→  // Reply form
    85→  const [replyTexts, setReplyTexts] = useState<Record<number, string>>({});
    86→
    87→  // Removing subject
    88→  const [removingId, setRemovingId] = useState<string | number | null>(null);
    89→
    90→  // Group chapters and lessons by subject
    91→  const subjectDetails = useMemo(() => {
    92→    return subjects.map((cs: any) => {
    93→      const subjId = cs.subjectId || cs.subject_id;
    94→      const subjName = cs.subjectName || cs.subject_name || cs.name || 'Unknown Subject';
    95→      const subjSlug = cs.subjectSlug || cs.subject_slug || cs.slug || '';
    96→      const subjTechId = cs.technologyId || cs.technology_id || 0;
    97→      const subjSemester = cs.semester || '';
    98→
    99→      const subjectChapters = (curriculum.chapters || []).filter(
   100→        (ch: any) => String(ch.subjectId || ch.subject_id) === String(subjId)
   101→      ).sort((a: any, b: any) => (a.sortOrder || a.sort_order || 0) - (b.sortOrder || b.sort_order || 0));
   102→
   103→      const chapterIds = subjectChapters.map((ch: any) => ch.id);
   104→      const subjectLessons = (curriculum.lessons || []).filter(
   105→        (l: any) => chapterIds.includes(l.chapterId || l.chapter_id) || String(l.subjectId || l.subject_id) === String(subjId)
   106→      );
   107→
   108→      const videoCount = subjectLessons.filter((l: any) => (l.lessonType || l.lesson_type) === 'video').length;
   109→      const docCount = subjectLessons.filter((l: any) => (l.lessonType || l.lesson_type) === 'document').length;
   110→      const liveCount = subjectLessons.filter((l: any) => (l.lessonType || l.lesson_type) === 'link').length;
   111→
   112→      return {
   113→        ...cs,
   114→        subjectId: subjId,
   115→        name: subjName,
   116→        slug: subjSlug,
   117→        technologyId: subjTechId,
   118→        semester: subjSemester,
   119→        chapters: subjectChapters,
   120→        lessons: subjectLessons,
   121→        chapterCount: subjectChapters.length,
   122→        lessonCount: subjectLessons.length,
   123→        videoCount,
   124→        docCount,
   125→        liveCount,
   126→      };
   127→    });
   128→  }, [subjects, curriculum]);
   129→
   130→  const toggleSubject = (subjId: string) => {
   131→    setExpandedSubjects((prev) => {
   132→      const next = new Set(prev);
   133→      if (next.has(String(subjId))) next.delete(String(subjId));
   134→      else next.add(String(subjId));
   135→      return next;
   136→    });
   137→  };
   138→
   139→  const toggleChapter = (chapterId: string) => {
   140→    setExpandedChapters((prev) => {
   141→      const next = new Set(prev);
   142→      if (next.has(chapterId)) next.delete(chapterId);
   143→      else next.add(chapterId);
   144→      return next;
   145→    });
   146→  };
   147→
   148→  const startAddChapter = (subjectId: string) => {
   149→    setAddMode('chapter');
   150→    setAddSubjectId(String(subjectId));
   151→    setAddChapterId('');
   152→    setChapterTitle('');
   153→    setChapterDesc('');
   154→  };
   155→
   156→  const startAddLesson = (subjectId: string, chapterId: string) => {
   157→    setAddMode('lesson');
   158→    setAddSubjectId(String(subjectId));
   159→    setAddChapterId(chapterId);
   160→    setLessonTitle('');
   161→    setLessonVideoUrl('');
   162→    setLessonDesc('');
   163→    setLessonIsPreview(false);
   164→    setLessonType('video');
   165→  };
   166→
   167→  const handleCreateChapter = async () => {
   168→    if (!chapterTitle.trim()) { toast.error('Enter chapter title'); return; }
   169→    const subject = subjectDetails.find((s: any) => String(s.subjectId) === addSubjectId);
   170→    const result = await createChapter({
   171→      title: chapterTitle.trim(),
   172→      subjectId: addSubjectId,
   173→      description: chapterDesc.trim() || undefined,
   174→      sortOrder: (subject?.chapters?.length || 0) + 1,
   175→    });
   176→    if (result) {
   177→      toast.success('Chapter created');
   178→      setAddMode(null);
   179→      refetchCurriculum();
   180→    } else {
   181→      toast.error('Failed to create chapter');
   182→    }
   183→  };
   184→
   185→  const handleDeleteChapter = async (chapterId: string) => {
   186→    if (!confirm('Delete this chapter and all its lessons?')) return;
   187→    const result = await deleteChapter({ chapterId });
   188→    if (result !== null) {
   189→      toast.success('Chapter deleted');
   190→      refetchCurriculum();
   191→    } else {
   192→      toast.error('Failed to delete chapter');
   193→    }
   194→  };
   195→
   196→  const handleCreateLesson = async () => {
   197→    if (!lessonTitle.trim()) { toast.error('Enter lesson title'); return; }
   198→    if (!addChapterId) { toast.error('Select a chapter'); return; }
   199→    const subject = subjectDetails.find((s: any) => String(s.subjectId) === addSubjectId);
   200→    const chapterLessons = (subject?.lessons || []).filter((l: any) => (l.chapterId || l.chapter_id) === addChapterId);
   201→    const result = await createLesson({
   202→      title: lessonTitle.trim(),
   203→      chapterId: addChapterId,
   204→      subjectId: addSubjectId as any,
   205→      lessonType,
   206→      videoUrl: lessonVideoUrl.trim() || undefined,
   207→      description: lessonDesc.trim() || undefined,
   208→      isPreview: lessonIsPreview,
   209→      sortOrder: chapterLessons.length + 1,
   210→    });
   211→    if (result) {
   212→      toast.success('Lesson added');
   213→      setAddMode(null);
   214→      refetchCurriculum();
   215→    } else {
   216→      toast.error('Failed to add lesson');
   217→    }
   218→  };
   219→
   220→  const handleDeleteLesson = async (lessonId: string) => {
   221→    if (!confirm('Delete this lesson?')) return;
   222→    const result = await deleteLesson({ lessonId });
   223→    if (result !== null) {
   224→      toast.success('Lesson deleted');
   225→      refetchCurriculum();
   226→    } else {
   227→      toast.error('Failed to delete lesson');
   228→    }
   229→  };
   230→
   231→  const handleRemoveSubject = async (subjectId: string | number) => {
   232→    if (!confirm('Remove this subject from the course? This will not delete the subject itself.')) return;
   233→    setRemovingId(subjectId);
   234→    const result = await removeSubject({ subjectId: typeof subjectId === 'string' ? subjectId : Number(subjectId) });
   235→    if (result !== null) {
   236→      toast.success('Subject removed');
   237→      refetchSubjects();
   238→      refetchCurriculum();
   239→    } else {
   240→      toast.error('Failed to remove subject');
   241→    }
   242→    setRemovingId(null);
   243→  };
   244→
   245→  const handleReply = async (reviewId: number) => {
   246→    const text = replyTexts[reviewId]?.trim();
   247→    if (!text) return;
   248→    const result = await replyReview({ reviewId, replyText: text });
   249→    if (result) {
   250→      toast.success('Reply posted');
   251→      setReplyTexts((prev) => ({ ...prev, [reviewId]: '' }));
   252→      refetchReviews();
   253→    } else {
   254→      toast.error('Failed to post reply');
   255→    }
   256→  };
   257→
   258→  // Course sub-navigation tabs
   259→  const courseTabs = [
   260→    { id: 'overview', label: 'Overview', icon: BookOpen, page: 'course-detail' as const },
   261→    { id: 'curriculum', label: 'Curriculum', icon: FolderOpen, page: 'course-curriculum' as const },
   262→    { id: 'videos', label: 'Videos', icon: Video, page: 'course-videos' as const },
   263→    { id: 'resources', label: 'Resources', icon: FileText, page: 'course-resources' as const },
   264→    { id: 'instructors', label: 'Instructors', icon: Users, page: 'course-instructors' as const },
   265→    { id: 'settings', label: 'Settings', icon: Radio, page: 'course-settings' as const },
   266→    { id: 'live', label: 'Live', icon: Radio, page: 'course-live' as const },
   267→  ];
   268→
   269→  if (courseLoading || subjectsLoading || curriculumLoading) {
   270→    return <LoadingSkeleton type="card" count={3} />;
   271→  }
   272→
   273→  return (
   274→    <div className="space-y-6">
   275→      {/* Header */}
   276→      <motion.div initial={{ opacity: 0, y: 8 }} animate={{ opacity: 1, y: 0 }} transition={{ duration: 0.3 }}>
   277→        <div className="flex items-center justify-between gap-4">
   278→          <div>
   279→            <h1 className="text-2xl md:text-3xl font-extrabold text-foreground">
   280→              <span className="gradient-text">Curriculum</span>
   281→            </h1>
   282→            <p className="text-muted-foreground mt-1 text-sm">{course?.title || ''}</p>
   283→          </div>
   284→          <GradientButton
   285→            size="sm"
   286→            icon={<Plus className="w-4 h-4" />}
   287→            onClick={() => navigate('course-add-subject', { courseId })}
   288→          >
   289→            Add Subject
   290→          </GradientButton>
   291→        </div>
   292→      </motion.div>
   293→
   294→      {/* Course Sub-Navigation */}
   295→      <motion.div
   296→        initial={{ opacity: 0, y: 8 }}
   297→        animate={{ opacity: 1, y: 0 }}
   298→        transition={{ delay: 0.1, duration: 0.3 }}
   299→      >
   300→        <div className="flex items-center gap-1 overflow-x-auto pb-1">
   301→          {courseTabs.map((tab) => (
   302→            <button
   303→              key={tab.id}
   304→              onClick={() => navigate(tab.page, { courseId })}
   305→              className={cn(
   306→                'relative flex items-center gap-2 px-4 py-2.5 text-sm font-semibold rounded-xl transition-colors whitespace-nowrap',
   307→                tab.id === 'curriculum'
   308→                  ? 'bg-black/[0.05] dark:bg-white/[0.06] text-foreground'
   309→                  : 'text-muted-foreground hover:text-foreground hover:bg-black/[0.02] dark:hover:bg-white/[0.02]'
   310→              )}
   311→            >
   312→              <tab.icon className="w-4 h-4" />
   313→              {tab.label}
   314→              {tab.id === 'curriculum' && (
   315→                <motion.div
   316→                  className="absolute bottom-0 left-2 right-2 h-[2px] rounded-full bg-black dark:bg-white"
   317→                  layoutId="curriculum-tab"
   318→                  transition={{ type: 'spring', stiffness: 350, damping: 30 }}
   319→                />
   320→              )}
   321→            </button>
   322→          ))}
   323→        </div>
   324→      </motion.div>
   325→
   326→      {/* Subjects List */}
   327→      {subjectDetails.length === 0 ? (
   328→        <EmptyState
   329→          icon={FolderOpen}
   330→          title="No subjects added yet"
   331→          description="Add subjects to organize your course content. Each subject can have chapters, lessons, videos, and more."
   332→          actionLabel="Add Subject"
   333→          onAction={() => navigate('course-add-subject', { courseId })}
   334→        />
   335→      ) : (
   336→        <div className="space-y-3">
   337→          {subjectDetails.map((subject: any, i: number) => {
   338→            const subjKey = String(subject.subjectId || subject.id);
   339→            const isSubjectExpanded = expandedSubjects.has(subjKey);
   340→
   341→            return (
   342→              <motion.div
   343→                key={subjKey || i}
   344→                initial={{ opacity: 0, y: 12 }}
   345→                animate={{ opacity: 1, y: 0 }}
   346→                transition={{ delay: i * 0.05, duration: 0.3 }}
   347→              >
   348→                <GlassCard className="p-0 overflow-hidden">
   349→                  {/* Subject Header - Clickable to expand */}
   350→                  <div
   351→                    className="p-4 md:p-5 cursor-pointer hover:bg-black/[0.01] dark:hover:bg-white/[0.01] transition-colors"
   352→                    onClick={() => toggleSubject(subjKey)}
   353→                  >
   354→                    <div className="flex items-start gap-3 md:gap-4">
   355→                      {/* Chevron Indicator */}
   356→                      <motion.div
   357→                        animate={{ rotate: isSubjectExpanded ? 0 : -90 }}
   358→                        transition={{ duration: 0.2 }}
   359→                        className="flex-shrink-0 mt-1"
   360→                      >
   361→                        <ChevronDown className="w-4 h-4 text-muted-foreground" />
   362→                      </motion.div>
   363→
   364→                      {/* Subject Icon */}
   365→                      <div className="w-10 h-10 md:w-12 md:h-12 rounded-2xl bg-gradient-to-br from-neutral-900 via-neutral-800 to-black dark:from-white dark:via-neutral-200 dark:to-white flex items-center justify-center shadow-[0_4px_15px_rgba(0,0,0,0.15)] flex-shrink-0">
   366→                        <BookOpen className="w-4 h-4 md:w-5 md:h-5 text-white dark:text-neutral-900" />
   367→                      </div>
   368→
   369→                      {/* Subject Info */}
   370→                      <div className="flex-1 min-w-0">
   371→                        <div className="flex items-center gap-2">
   372→                          <h3 className="text-base font-bold text-foreground truncate">{subject.name}</h3>
   373→                          {subject.semester && (
   374→                            <span className="text-[10px] font-bold px-2 py-0.5 rounded-full bg-neutral-100 text-neutral-700 dark:bg-neutral-800/30 dark:text-neutral-300">
   375→                              {subject.semester}
   376→                            </span>
   377→                          )}
   378→                        </div>
   379→
   380→                        {/* Stats row */}
   381→                        <div className="flex items-center gap-3 mt-1.5 flex-wrap">
   382→                          {subject.chapterCount > 0 && (
   383→                            <span className="flex items-center gap-1 text-xs text-muted-foreground">
   384→                              <FolderOpen className="w-3 h-3" />
   385→                              {subject.chapterCount} {subject.chapterCount === 1 ? 'chapter' : 'chapters'}
   386→                            </span>
   387→                          )}
   388→                          {subject.lessonCount > 0 && (
   389→                            <span className="flex items-center gap-1 text-xs text-muted-foreground">
   390→                              <FileText className="w-3 h-3" />
   391→                              {subject.lessonCount} {subject.lessonCount === 1 ? 'lesson' : 'lessons'}
   392→                            </span>
   393→                          )}
   394→                          {subject.videoCount > 0 && (
   395→                            <span className="flex items-center gap-1 text-xs text-muted-foreground">
   396→                              <Video className="w-3 h-3" />
   397→                              {subject.videoCount} {subject.videoCount === 1 ? 'video' : 'videos'}
   398→                            </span>
   399→                          )}
   400→                          {subject.docCount > 0 && (
   401→                            <span className="flex items-center gap-1 text-xs text-muted-foreground">
   402→                              <StickyNote className="w-3 h-3" />
   403→                              {subject.docCount} {subject.docCount === 1 ? 'document' : 'documents'}
   404→                            </span>
   405→                          )}
   406→                          {subject.liveCount > 0 && (
   407→                            <span className="flex items-center gap-1 text-xs text-muted-foreground">
   408→                              <Radio className="w-3 h-3" />
   409→                              {subject.liveCount} {subject.liveCount === 1 ? 'link' : 'links'}
   410→                            </span>
   411→                          )}
   412→                          {subject.chapterCount === 0 && subject.lessonCount === 0 && (
   413→                            <span className="text-xs text-muted-foreground/60 italic">No content yet</span>
   414→                          )}
   415→                        </div>
   416→                      </div>
   417→
   418→                      {/* Actions */}
   419→                      <div className="flex items-center gap-1 flex-shrink-0" onClick={(e) => e.stopPropagation()}>
   420→                        <button
   421→                          onClick={() => startAddChapter(subjKey)}
   422→                          className="w-8 h-8 rounded-lg flex items-center justify-center text-muted-foreground hover:text-foreground hover:bg-black/[0.04] dark:hover:bg-white/[0.04] transition-colors"
   423→                          title="Add chapter"
   424→                        >
   425→                          <Plus className="w-3.5 h-3.5" />
   426→                        </button>
   427→                        <button
   428→                          onClick={() => handleRemoveSubject(subject.subjectId || subject.id)}
   429→                          disabled={removingId === (subject.subjectId || subject.id)}
   430→                          className="w-8 h-8 rounded-lg flex items-center justify-center text-muted-foreground hover:text-red-500 hover:bg-red-50 dark:hover:bg-red-900/20 transition-colors"
   431→                        >
   432→                          <Trash2 className="w-3.5 h-3.5" />
   433→                        </button>
   434→                      </div>
   435→                    </div>
   436→                  </div>
   437→
   438→                  {/* Expanded Subject Content */}
   439→                  <AnimatePresence>
   440→                    {isSubjectExpanded && (
   441→                      <motion.div
   442→                        initial={{ height: 0, opacity: 0 }}
   443→                        animate={{ height: 'auto', opacity: 1 }}
   444→                        exit={{ height: 0, opacity: 0 }}
   445→                        transition={{ duration: 0.3, ease: [0.4, 0, 0.2, 1] }}
   446→                        className="overflow-hidden"
   447→                      >
   448→                        <div className="border-t border-black/[0.03] dark:border-white/[0.03] bg-black/[0.005] dark:bg-white/[0.005]">
   449→                          {/* Inline Add Chapter Form */}
   450→                          <AnimatePresence>
   451→                            {addMode === 'chapter' && addSubjectId === subjKey && (
   452→                              <motion.div
   453→                                initial={{ height: 0, opacity: 0 }}
   454→                                animate={{ height: 'auto', opacity: 1 }}
   455→                                exit={{ height: 0, opacity: 0 }}
   456→                                transition={{ duration: 0.25 }}
   457→                                className="overflow-hidden"
   458→                              >
   459→                                <div className="p-4 m-3 rounded-xl bg-white/60 dark:bg-neutral-800/40 border border-black/[0.04] dark:border-white/[0.04]">
   460→                                  <div className="flex items-center justify-between mb-3">
   461→                                    <h4 className="text-sm font-bold gradient-text">New Chapter</h4>
   462→                                    <button onClick={() => setAddMode(null)} className="w-7 h-7 rounded-lg bg-muted/50 flex items-center justify-center text-muted-foreground hover:text-foreground transition-colors">
   463→                                      <X className="w-3.5 h-3.5" />
   464→                                    </button>
   465→                                  </div>
   466→                                  <div className="space-y-2.5">
   467→                                    <input
   468→                                      type="text"
   469→                                      value={chapterTitle}
   470→                                      onChange={(e) => setChapterTitle(e.target.value)}
   471→                                      placeholder="Chapter title *"
   472→                                      className="w-full px-3 py-2 rounded-lg text-sm premium-input placeholder:text-muted-foreground/60"
   473→                                      autoFocus
   474→                                    />
   475→                                    <textarea
   476→                                      value={chapterDesc}
   477→                                      onChange={(e) => setChapterDesc(e.target.value)}
   478→                                      placeholder="Optional description"
   479→                                      rows={2}
   480→                                      className="w-full px-3 py-2 rounded-lg text-sm premium-input placeholder:text-muted-foreground/60 resize-none"
   481→                                    />
   482→                                    <div className="flex justify-end gap-2">
   483→                                      <GradientButton variant="ghost" size="sm" onClick={() => setAddMode(null)}>Cancel</GradientButton>
